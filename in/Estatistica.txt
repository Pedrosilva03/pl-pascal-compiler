program Estatisticas;

function MediaTotal(soma: integer): real;
begin
  MediaTotal := soma / 5;
end;

var
  valores: array[1..5] of integer;
  i, min, max, soma: integer;
  media: real;
  comp1: integer;
begin
  writeln('Introduza 5 números:');
  
  for i := 1 to 5 do
    readln(valores[i]);
  
  min := valores[1];
  max := valores[1];
  soma := 0;

  for i := 1 to 5 do
  begin
    comp1 := valores[i];
    if comp1 < min then
      min := valores[i];
    if comp1 > max then
      max := valores[i];
    soma := soma + valores[i];
  end;

  media := MediaTotal(soma);

  writeln('Mínimo: ', min);
  writeln('Máximo: ', max);
  writeln('Média: ', media);
end.
